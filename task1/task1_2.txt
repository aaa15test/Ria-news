//создание БД
CREATE DATABASE authors;

USE authors

//создадим таблицу author_id, где первое поле - id автора (первичный ключ), 
//а второе поле - имя автора
CREATE TABLE author_id (id int(20) AUTO_INCREMENT, author_name VARCHAR(20) NOT NULL, PRIMARY KEY(id));

//заполним таблицу author_id данными из файла author_id.txt

//создадим таблицу book_id, где первое поле - id книги (первичный ключ), второе - название книги.
//Поле id_book является внешним ключом, т.е. оно может иметь только те значения, которые есть в поле id таблицы author,
//поэтому для него используется FOREIGN KEY
CREATE TABLE book_id (id_book int(20) AUTO_INCREMENT, book_name VARCHAR(30) NOT NULL, PRIMARY KEY(id_book), FOREIGN KEY (id_book) REFERENCES author_id (id));
